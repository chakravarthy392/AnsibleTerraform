---
  - name: SImple play to install httpd
    hosts: all
    gather_facts: true
    become: yes
    tasks:
      - name: Installing httpd using yum
        yum: name=httpd state=latest
        when: ansible_distribution != "Ubuntu"
      - name: Istalling apache2 using apt
        apt: name=apache2 state=present
        when: ansible_distribution == "Ubuntu"
